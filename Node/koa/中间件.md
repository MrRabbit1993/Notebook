#### 一、中间件
可以使用多个中间件，默认会传递ctx和next 值，有下一个中间件，需要调取next
```js 
const Koa = require('koa');
const app = new Koa();
app.use(async (ctx,next) => {//注册中间件
    console.log(1);
    await next()
});
app.use(async (ctx,next) => {
    console.log(2);
    await next();
});
app.listen(3000);
```
