#### 一、匹陪文件后缀
```js
const fileReg = /\.([0-9a-z]+)(?:[\?#]|$)/i;
const filesStr = "a.jpg";
const filesStr1 = "a.jpg.png";
const res = fileReg.match(filesStr);//[".jpg", "jpg", index: 1, input: "a.jpg", groups: undefined]
const res = filesStr1.match(filesStr);//[".png", "png", index: 5, input: "a.jpg.png", groups: undefined]
```
